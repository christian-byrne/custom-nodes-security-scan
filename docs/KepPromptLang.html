14	def queue_prompt(prompt):
15	    p = {"prompt": prompt, "client_id": client_id}
16	    data = json.dumps(p).encode('utf-8')
17	    req =  urllib.request.Request("http://{}/prompt".format(server_address), data=data)
18	
19	    try:
20	        response = urllib.request.urlopen(req)

22	    except urllib.error.HTTPError as e:
23	        print(f"HTTP Error {e.code}: {e.reason}")
24	        error_body = e.read()

26	        try:
27	            json_error_body = json.loads(error_body)

29	            raise e

31	            print("Failed to decode error response as JSON.")

34	    except Exception as e:
35	        print(f"An unexpected error occurred: {e}")

48	def get_images(ws, prompt):
49	    prompt_id = queue_prompt(prompt)['prompt_id']
50	    output_images = {}
51	    while True:
52	        out = ws.recv()
53	        if isinstance(out, str):
54	            message = json.loads(out)
55	            if message['type'] == 'executing':
56	                data = message['data']
57	                if data['node'] is None and data['prompt_id'] == prompt_id:
58	                    break #Execution is done
59	        else:
60	            continue #previews are binary data
61	
62	    history = get_history(prompt_id)[prompt_id]
63	    for o in history['outputs']:
64	        for node_id in history['outputs']:
65	            node_output = history['outputs'][node_id]
66	            if 'images' in node_output:
67	                images_output = []
68	                for image in node_output['images']:
69	                    image_data = get_image(image['filename'], image['subfolder'], image['type'])
70	                    images_output.append(image_data)
71	            output_images[node_id] = images_output
72	
73	    return output_images
