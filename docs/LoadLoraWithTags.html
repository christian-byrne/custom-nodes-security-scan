37	def calculate_sha256(file_path):
38	    sha256_hash = hashlib.sha256()
39	    with open(file_path, "rb") as f:
40	        for chunk in iter(lambda: f.read(4096), b""):
41	            sha256_hash.update(chunk)
42	    return sha256_hash.hexdigest()
